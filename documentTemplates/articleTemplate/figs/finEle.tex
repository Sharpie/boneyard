\begin{tikzpicture}[>=stealth,
	emNode/.style={circle,draw,ultra thick,fill=white,scale=0.75},
	alphNode/.style={circle,draw,thick,text=white,fill=black,minimum size=8mm,scale=0.75}]
	
	\matrix (nums) [matrix of nodes,nodes=emNode,column sep=2cm] at (0,0)
	{
		1 & 2 & 3 & 4 \\
	};

	\foreach \x / \y in {1/2,2/3,3/4} { 
		\draw[ultra thick] (nums-1-\x) -- (nums-1-\y);
	}
	
	\matrix (alphs) [matrix of nodes,nodes=alphNode,column sep=2cm] at (0,0.75)
	{
		{\bf a} & {\bf b} & {\bf c}  \\
	};
	
	
	\begin{scope}[decoration=snake,segment length=1cm]
		\draw[ultra thick,color=blue,decorate] (nums-1-1) -- +(-2,0);
		\draw[ultra thick,color=blue,decorate] (nums-1-4) -- +(2,0);
	\end{scope}
	
	\begin{scope}[decoration={brace,mirror,raise=0.25cm,amplitude=0.25cm}]
		\draw[rotate=180,decorate,thick] (nums-1-1.south) -- 
		node[below=0.5cm] {Node Numbers}
		(nums-1-4.south);
	\end{scope}
	
	\begin{scope}[decoration={brace,raise=0.25cm,amplitude=0.25cm}]
		\draw[rotate=180,decorate,thick] (alphs-1-1.north) -- 
		node[above=0.5cm] {Element Names}
		(alphs-1-3.north);
	\end{scope}
	
\end{tikzpicture}
